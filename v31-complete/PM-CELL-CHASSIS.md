### OutboxPublisherService (BackgroundService)

```csharp
// src/Cells/[CellName]/AHS.Cell.[CellName].Infrastructure/Services/OutboxPublisherService.cs
namespace AHS.Cell.[CellName].Infrastructure.Services;

public sealed class OutboxPublisherService(
    IServiceScopeFactory                    scopeFactory,
    ServiceBusSender                        sender,
    ILogger<OutboxPublisherService>         log)
    : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            await PublishPendingAsync(ct);
            await Task.Delay(TimeSpan.FromSeconds(10), ct);
        }
    }

    private async Task PublishPendingAsync(CancellationToken ct)
    {
        await using var scope = scopeFactory.CreateAsyncScope();
        var conn = scope.ServiceProvider
            .GetRequiredService<IDbConnectionFactory>();

        await using var db = await conn.CreateAsync(ct);

        var pending = (await db.QueryAsync<OutboxMessage>("""
            SELECT id, tenant_id, event_type, payload_json,
                   topic, created_at, retry_count
            FROM outbox_messages
            WHERE published_at IS NULL
              AND retry_count < 5
            ORDER BY created_at
            LIMIT 50
            FOR UPDATE SKIP LOCKED
            """)).AsList();

        foreach (var msg in pending)
        {
            try
            {
                var sbMessage = new ServiceBusMessage(msg.PayloadJson)
                {
                    Subject       = msg.EventType,
                    ContentType   = "application/json",
                    MessageId     = msg.Id.ToString(),
                    CorrelationId = msg.TenantId.ToString()
                };

                await sender.SendMessageAsync(sbMessage, ct);

                await db.ExecuteAsync("""
                    UPDATE outbox_messages
                    SET published_at = NOW(), error = NULL
                    WHERE id = @id
                    """, new { id = msg.Id });

                log.LogInformation(
                    "[CellName] Outbox: published {EventType} tenant {TenantId}",
                    msg.EventType, msg.TenantId);
            }
            catch (Exception ex)
            {
                await db.ExecuteAsync("""
                    UPDATE outbox_messages
                    SET retry_count = retry_count + 1, error = @error
                    WHERE id = @id
                    """, new { id = msg.Id, error = ex.Message });

                log.LogWarning(ex,
                    "[CellName] Outbox: failed {EventType} retry {Retry}",
                    msg.EventType, msg.RetryCount + 1);
            }
        }
    }
}
```

    string                  Status,
    [CellName]LifecycleState Lifecycle,

public sealed class [CellName]HealthService(IDbConnectionFactory db)

    public async Task<[CellName]HealthDto> GetHealthAsync(CancellationToken ct)

            Lifecycle:    lifecycle,
