public RiskScore Aggregate(ReadOnlySpan<double> factors)
{
    Span<double> weighted = factors.Length <= 14
        ? stackalloc double[factors.Length]
        : new double[factors.Length];

    for (int i = 0; i < factors.Length; i++)
        weighted[i] = factors[i] * _weights[i];

    double total = 0;
    foreach (var w in weighted) total += w;
    return new RiskScore(Math.Min(100, total));

        Status: "Acceptable",

public class HistoricalRepository(IDbConnectionFactory db)
    : IHistoricalRepository
{
    public async Task<IReadOnlyList<HistoricalRecordDto>> SelectCompatibleAsync(
        string   routeId,
        string   packagingType,
        string   season,
        string?  carrierId,     // optional — carrier is a modifier
        int      minRecords,
        int      maxRecords,

            ? """

              WHERE route_id = @routeId
                AND packaging_type = @packagingType
                AND season = @season
                AND carrier_id = @carrierId
              ORDER BY recorded_at DESC
              LIMIT @maxRecords
              """
            : """
              SELECT id, route_id, packaging_type, carrier_id, season,
                     excursion_occurred, risk_score, recorded_at
              FROM historical_records
              WHERE route_id = @routeId
                AND packaging_type = @packagingType
                AND season = @season
              ORDER BY recorded_at DESC
              LIMIT @maxRecords
              """;

// Infrastructure/DomainServices/ReadinessValidator.cs
// Implements IReadinessValidator (Domain port)
public class ReadinessValidator(IHistoricalRepository history)
    : IReadinessValidator
{
    public async Task<ReadinessResult> ValidateAsync(
        ShipmentInputDto input, CancellationToken ct)
    {
        var errors   = new List<string>();
        var warnings = new List<string>();

        // Check 1: Carrier historical problems
        if (input.CarrierIncidents12M > 5)
            warnings.Add($"CARRIER_HIGH_INCIDENTS: {input.CarrierId} " +
                        $"has {input.CarrierIncidents12M} incidents in 12 months");

        // Check 2: Season incompatibility
        var season = ShipmentIdentity.GetSeason(input.DepartureDate);
        if (input.ForecastMaxCelsius > 35 && input.PackagingType == "PassiveChamber")
            errors.Add("SEASON_INCOMPATIBLE: PassiveChamber in extreme heat forecast");

        // Check 3: Route divergence check
        var knownRoutes = await history.GetKnownRoutesAsync(ct);
        if (!knownRoutes.Contains(input.RouteId))
            warnings.Add($"ROUTE_UNSEEN: {input.RouteId} has no historical data");

        // Check 4: Dataset size
        var count = await history.CountCompatibleAsync(
            input.RouteId, input.PackagingType, season, ct);
        if (count < 5)
            errors.Add($"DATASET_TOO_SMALL: Only {count} records for this profile");
        if (count > 10_000)
            warnings.Add("DATASET_LARGE: Consider pruning historical data");

        // Check 5: Thermal outliers
        if (input.ForecastMaxCelsius - input.ForecastMinCelsius > 40)
            warnings.Add("THERMAL_OUTLIER: Extreme temperature spread detected");

        // Check 6: Packaging incompatibility
        if (input.ProductCategory == "Pharmaceutical" &&
            input.PackagingType == "PassiveChamber" &&
            input.EstimatedDurationHours > 48)
            errors.Add("PACKAGING_INCOMPATIBLE: Pharma + Passive > 48h exceeds safe window");

        // Check 7: New carrier/route not seen before
        var newCarrier = !await history.CarrierExistsAsync(input.CarrierId, ct);
        if (newCarrier)
            warnings.Add($"NEW_CARRIER: {input.CarrierId} has no historical baseline");

        // Check 8: Abrupt climate change
        if (Math.Abs(input.ForecastMaxCelsius - input.ForecastMinCelsius) > 30)
            warnings.Add("CLIMATE_ABRUPT_CHANGE: High forecast variance detected");

        // Check 9: Missing values (validated by FluentValidation in API layer)
        if (string.IsNullOrWhiteSpace(input.RouteId))
            errors.Add("MISSING_VALUE: RouteId is required");

        // Return typed ReadinessState — DDD-correct pattern matching
        if (errors.Count > 0)
            return new NotAcceptableState(
                Issues: [..errors],
                Recommendations: [..warnings]);

        if (warnings.Count > 3)
            return new RiskyState(
                Explanation: $"Multiple risk indicators detected ({warnings.Count} warnings)",
                Warnings: [..warnings]);

        return new AcceptableState(Warnings: [..warnings]);
    }
}

// Infrastructure/DomainServices/DivergenceDetector.cs
public class DivergenceDetector(IHistoricalRepository history)
    : IDivergenceDetector
{
    // Detects structural differences between current shipment and historical baseline
    // Returns true if significant divergence found (triggers retrain consideration)
    public async Task<DivergenceReport> DetectAsync(
        ShipmentInputDto input, CancellationToken ct)
    {
        var divergences = new List<string>();

        // Route divergence
        var avgRiskByRoute = await history.GetAvgRiskByRouteAsync(input.RouteId, ct);
        if (avgRiskByRoute is null)
            divergences.Add($"ROUTE_NO_BASELINE: {input.RouteId}");

        // Carrier divergence (modifier — not identity)
        var carrierBaseline = await history.GetCarrierBaselineAsync(input.CarrierId, ct);
        if (carrierBaseline is null)
            divergences.Add($"CARRIER_NO_BASELINE: {input.CarrierId}");

        // Packaging divergence
        var pkgCompatible = await history.HasCompatiblePackagingAsync(
            input.RouteId, input.PackagingType, ct);
        if (!pkgCompatible)
            divergences.Add($"PACKAGING_MISMATCH: No history for " +
                           $"{input.PackagingType} on {input.RouteId}");

        // Seasonal divergence
        var season = ShipmentIdentity.GetSeason(input.DepartureDate);
        var seasonalCount = await history.CountCompatibleAsync(
            input.RouteId, input.PackagingType, season, ct);
        if (seasonalCount < 3)
            divergences.Add($"SEASON_UNDERREPRESENTED: Only {seasonalCount} " +
                           $"records for {season}");

        return new DivergenceReport(
            HasDivergence: divergences.Count > 0,
            Divergences: [..divergences]);
    }
}

// Infrastructure/DomainServices/HistoricalSelector.cs
// Selection criteria: route + packaging + season (carrier optional)
public class HistoricalSelector(IHistoricalRepository history)
    : IHistoricalSelector
{
    private const int MinRecords = 5;
    private const int MaxRecords = 500;

    public async Task<HistoricalDataset> SelectAsync(
        ShipmentInputDto input,
        bool             includeCarrier,  // user decision
        CancellationToken ct)
    {
        var season = ShipmentIdentity.GetSeason(input.DepartureDate);

        var records = await history.SelectCompatibleAsync(
            routeId:       input.RouteId,
            packagingType: input.PackagingType,
            season:        season,
            carrierId:     includeCarrier ? input.CarrierId : null,
            minRecords:    MinRecords,
            maxRecords:    MaxRecords,
            ct);

        // Remove outliers (IQR method — no LINQ, direct array manipulation)
        var cleaned = RemoveOutliers(records);

        return new HistoricalDataset(
            Records:    cleaned,
            Season:     season,
            RecordCount: cleaned.Count,
            IncludedCarrier: includeCarrier ? input.CarrierId : null
        );
    }

    private static IReadOnlyList<HistoricalRecordDto> RemoveOutliers(
        IReadOnlyList<HistoricalRecordDto> records)
    {
        if (records.Count < 4) return records;

        // Calculate IQR on risk_score without LINQ
        var scores = new double[records.Count];
        for (int i = 0; i < records.Count; i++)
            scores[i] = records[i].RiskScore ?? 0;

        Array.Sort(scores);
        var q1  = scores[scores.Length / 4];
        var q3  = scores[scores.Length * 3 / 4];
        var iqr = q3 - q1;
        var lo  = q1 - 1.5 * iqr;
        var hi  = q3 + 1.5 * iqr;

        var result = new List<HistoricalRecordDto>(records.Count);
        foreach (var r in records)
            if (r.RiskScore >= lo && r.RiskScore <= hi)
                result.Add(r);

        return result;
    }
}

public class RetrainDecider(IModelRepository models)
    : IRetrainDecider
{
    public async Task<RetrainDecision> EvaluateAsync(
        HistoricalDataset dataset,
        DivergenceReport  divergence,
        CancellationToken ct)
    {
        var activeModel = await models.GetActiveAsync(ct);

        // Criterion 1: No model exists
        if (activeModel is null)
            return new RetrainDecision(true, "NO_MODEL_EXISTS", "Critical");

        // Criterion 2: +N shipments since last retrain
        var newSinceRetrain = await models.CountSinceLastRetrainAsync(ct);
        if (newSinceRetrain >= 50)
            return new RetrainDecision(true,
                $"THRESHOLD_EXCEEDED: {newSinceRetrain} new records", "High");

        // Criterion 3: New carriers or routes (divergence detected)
        if (divergence.HasDivergence)
            return new RetrainDecision(true,
                $"DIVERGENCE: {string.Join(", ", divergence.Divergences)}", "Medium");

        // Criterion 4: Seasonal shift
        var lastTrainSeason = GetSeason(activeModel.TrainedAt);
        var currentSeason   = GetSeason(DateTimeOffset.UtcNow);
        if (lastTrainSeason != currentSeason)
            return new RetrainDecision(true,
                $"SEASONAL_SHIFT: {lastTrainSeason}→{currentSeason}", "Medium");

        // Criterion 5: Model degradation (accuracy below threshold)
        if (activeModel.AccuracyScore < 0.75)
            return new RetrainDecision(true,
                $"MODEL_DEGRADED: accuracy={activeModel.AccuracyScore:F2}", "High");

        // Criterion 6: Time threshold (>30 days since last retrain)
        if (DateTimeOffset.UtcNow - activeModel.TrainedAt > TimeSpan.FromDays(30))
            return new RetrainDecision(true, "TEMPORAL_THRESHOLD: >30 days", "Low");

        return new RetrainDecision(false, "MODEL_CURRENT", "Low");
    }

    private static string GetSeason(DateTimeOffset date) => date.Month switch
    {
        12 or 1 or 2 => "Winter",
        3  or 4 or 5 => "Spring",
        6  or 7 or 8 => "Summer",

// Infrastructure/DomainServices/PredictionEngine.cs
// Weights: Route 40% + Carrier 30% + Environmental 30% + Passive penalty 15%
public class PredictionEngine : IPredictionEngine
{
    // XAI DNA — 14 diagnostic factors (fixed array — AOT safe)
    private static readonly string[] FactorNames =
    [
        "ROUTE_THREAT_LEVEL",        // 1
        "CARRIER_RELIABILITY",       // 2
        "EXT_TEMP_MAX_EXPOSURE",     // 3
        "EXT_TEMP_MIN_DELTA",        // 4
        "PACKAGING_INSULATION_PHI",  // 5
        "TRANSIT_DURATION_RISK",     // 6
        "DEPARTURE_HOUR_FACTOR",     // 7
        "SEASON_RISK_MODIFIER",      // 8
        "HISTORICAL_EXCURSION_RATE", // 9
        "CARRIER_INCIDENT_RATE",     // 10
        "HUMIDITY_STRESS_INDEX",     // 11
        "ROUTE_VARIANCE_SCORE",      // 12
        "TEMPORAL_DECAY_FACTOR",     // 13
        "GXP_COMPLIANCE_SCORE"       // 14
    ];

    public async ValueTask<PredictionResultDto> PredictAsync(
        ShipmentInputDto  input,
        HistoricalDataset dataset,
        ModelVersionEntity? model,
        CancellationToken ct)
    {
        // Calculate 14 factors — Span<double>, no heap allocations
        Span<double> factors = stackalloc double[14];

        factors[0]  = CalculateRouteThreat(input.RouteId, dataset);
        factors[1]  = 1.0 - input.CarrierReliabilityScore;
        factors[2]  = NormalizeTemp(input.ForecastMaxCelsius);
        factors[3]  = Math.Abs(input.ForecastMaxCelsius - input.ForecastMinCelsius) / 50.0;
        factors[4]  = input.PackagingType == "PassiveChamber" ? 0.85 : 0.40;
        factors[5]  = Math.Min(1.0, input.EstimatedDurationHours / 72.0);
        factors[6]  = input.DepartureHour is >= 12 and <= 16 ? 0.75 : 0.30;
        factors[7]  = GetSeasonRisk(input.DepartureDate);
        factors[8]  = CalculateExcursionRate(dataset);
        factors[9]  = Math.Min(1.0, input.CarrierIncidents12M / 10.0);
        factors[10] = Math.Min(1.0, input.ForecastHumidityPct / 100.0);
        factors[11] = CalculateRouteVariance(dataset);
        factors[12] = CalculateTemporalDecay(dataset);
        factors[13] = 1.0 - (dataset.RecordCount / 500.0);  // more data = more confidence

        // Weighted aggregation — no LINQ
        double routeScore       = factors[0] * 0.40;
        double carrierScore     = factors[1] * 0.30;
        double environmentScore = (factors[2] + factors[3]) / 2.0 * 0.30;
        double baseScore        = (routeScore + carrierScore + environmentScore) * 100;

        // Passive penalty: +15% base penalty (Blueprint REQ-001)
        if (input.PackagingType == "PassiveChamber")
            baseScore *= 1.15;

        var finalScore = Math.Min(100, baseScore);

        // Pessimistic TTF
        var physicalTtf    = EstimatePhysicalTtf(input);
        var pessimisticTtf = physicalTtf * (1 - finalScore / 100.0 * 0.60);
        var safeWindow     = pessimisticTtf * 0.80;

        // Build XAI DNA — 14 factors
        var dnaFactors = new XaiFactorDto[14];
        for (int i = 0; i < 14; i++)
            dnaFactors[i] = new XaiFactorDto(
                FactorId:     i + 1,
                Name:         FactorNames[i],
                Weight:       factors[i],
                Contribution: factors[i] * (i < 3 ? 0.40 : i < 6 ? 0.30 : 0.15),
                Narrative:    BuildNarrative(i, factors[i])
            );

        return new PredictionResultDto(
            PredictionId:            Guid.NewGuid(),
            CalculatedAt:            DateTimeOffset.UtcNow,
            RiskScore:               finalScore,
            RiskLevel:               finalScore >= 75 ? "Critical"
                                   : finalScore >= 50 ? "Elevated" : "Nominal",
            PessimisticTtfHours:     pessimisticTtf,
            SafeWindowHours:         safeWindow,
            XaiDna:                  new XaiDnaDto([..dnaFactors]),
            Recommendation:          BuildRecommendation(finalScore, input),
            RequiresImmediateAction: finalScore >= 75 || pessimisticTtf < 4
        );
    }
}
