```csharp
// Domain/Aggregates/Asset.cs
public sealed class Asset : AggregateRoot
{
    public Guid         Id              { get; private set; }
    public Guid         TenantId        { get; private set; }
    public string       SerialNumber    { get; private set; } = default!;
    public string       Name            { get; private set; } = default!;
    public AssetCategory Category       { get; private set; }
    public AssetStatus  Status          { get; private set; }
    public CalibrationRecord? LastCalibration { get; private set; }
    public MaintenanceRecord? LastMaintenance { get; private set; }
    public RiskFlag?    ActiveRiskFlag  { get; private set; }

    private Asset() { } // AOT-safe private ctor

    public static Asset Register(
        string        serialNumber,
        string        name,
        AssetCategory category,
        Guid          tenantId,
        SignedCommand  command)
    {
        var asset = new Asset
        {
            Id           = Guid.NewGuid(),
            TenantId     = tenantId,
            SerialNumber = serialNumber,
            Name         = name,
            Category     = category,
            Status       = AssetStatus.Active
        };
        asset.AddDomainEvent(new AssetRegistered(
            asset.Id, tenantId, serialNumber, name, category));
        return asset;
    }

    public void RecordMaintenance(
        string       technician,
        string       notes,
        SignedCommand command)
    {
        if (Status == AssetStatus.Retired)
            throw new AssetRetiredException(Id);

        LastMaintenance = new MaintenanceRecord(
            DateTimeOffset.UtcNow, technician, notes);
        Status = AssetStatus.Active;
        AddDomainEvent(new AssetMaintenanceRecorded(
            Id, TenantId, technician, DateTimeOffset.UtcNow));
    }

    public void FlagAtRisk(
        Guid   xinferPredictionId,
        double riskScore,
        string riskLevel)
    {
        ActiveRiskFlag = new RiskFlag(
            xinferPredictionId, riskScore, riskLevel, DateTimeOffset.UtcNow);
        AddDomainEvent(new AssetFlaggedAtRisk(
            Id, TenantId, xinferPredictionId, riskScore, riskLevel));
    }

    public void Retire(string reason, SignedCommand command)
    {
        if (Status == AssetStatus.Retired)
            throw new AssetAlreadyRetiredException(Id);

        Status = AssetStatus.Retired;
        AddDomainEvent(new AssetRetired(Id, TenantId, reason));
    }

    public static new Asset Rehydrate(IEnumerable<DomainEvent> history)
    {
        var asset = new Asset();
        ((AggregateRoot)asset).Rehydrate(history);
        return asset;
    }
}
```

```csharp
// RegisterAssetHandler
public class RegisterAssetHandler(
    IAssetRepository  repo,
    IOutboxWriter     outbox,
    IAssetManagerDbContext db)
{
    public async Task<Guid> HandleAsync(
        RegisterAssetCommand cmd, CancellationToken ct)
    {
        var asset = Asset.Register(
            cmd.SerialNumber, cmd.Name, cmd.Category,
            cmd.TenantId, cmd);

        await repo.SaveAsync(asset, ct);

        // Write to Outbox — atomic with DB save
        await outbox.WriteAsync(
            cmd.TenantId,
            nameof(AssetRegisteredEvent),
            JsonSerializer.Serialize(
                new AssetRegisteredEvent(
                    Guid.NewGuid(), cmd.TenantId.ToString(),
                    DateTimeOffset.UtcNow, asset.Id, asset.SerialNumber),
                AssetManagerContractsJsonContext.Default.AssetRegisteredEvent),
            ct);

        await db.SaveChangesAsync(ct);  // atomic: asset + outbox
        return asset.Id;
    }
}

// XinferEventConsumer — reacts to PredictOkEvent from Xinfer
// Runs as BackgroundService, subscribes to ahs.xinfer.events
public class XinferEventConsumer(
    IServiceScopeFactory scopeFactory,
    ServiceBusProcessor  processor,
    ILogger<XinferEventConsumer> log)
    : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        processor.ProcessMessageAsync += async args =>
        {
            var json    = args.Message.Body.ToString();
            var @event  = JsonSerializer.Deserialize(json,
                XinferContractsJsonContext.Default.PredictOkEvent)!;

            if (@event.RequiresImmediateAction)
            {
                await using var scope = scopeFactory.CreateAsyncScope();
                // Flag all assets associated with this shipment carrier
                // Implementation: query assets by carrier, call FlagAtRisk()
                log.LogWarning(
                    "AssetManager: Xinfer PredictOk — flagging assets. " +
                    "RiskScore={Score}", @event.RiskScore);
            }

            await args.CompleteMessageAsync(args.Message, ct);
        };

        await processor.StartProcessingAsync(ct);
        await Task.Delay(Timeout.Infinite, ct);
        await processor.StopProcessingAsync();
    }
}
```

```csharp
public class AssetRepository(
    IAssetManagerDbContext db,
    IDbConnectionFactory   dapper)
    : IAssetRepository
{
    // EF Core for writes
    public async Task SaveAsync(Asset asset, CancellationToken ct)
    {
        var existing = await db.Assets
            .FirstOrDefaultAsync(a => a.Id == asset.Id, ct);

        if (existing is null) await db.Assets.AddAsync(asset, ct);
        // SaveChangesAsync called by handler
    }

    // Dapper for reads
    public async Task<Asset?> GetByIdAsync(Guid id, CancellationToken ct)
    {
        await using var conn = await dapper.CreateAsync(ct);
        var row = await conn.QuerySingleOrDefaultAsync<AssetRow>("""
            SELECT id, tenant_id, serial_number, name, category,
                   status, risk_score, risk_level, flagged_at
            FROM assets WHERE id = @id
            """, new { id });

        return row is null ? null : MapToDomain(row);
    }
}
```

```csharp
// Endpoints/AssetEndpoints.cs
public static class AssetEndpoints
{
    public static void Map(WebApplication app)
    {
        var group = app.MapGroup("/api/assets")
            .RequireAuthorization();

        group.MapPost("/", async (
            RegisterAssetRequest req,
            RegisterAssetHandler handler,
            ITenantContext tenant,
            CancellationToken ct) =>
        {
            var id = await handler.HandleAsync(new RegisterAssetCommand(
                req.SerialNumber, req.Name, req.Category,
                tenant.TenantId,
                req.ReasonForChange, req.SignedById, req.SignedByName), ct);
            return Results.Created($"/api/assets/{id}", new { id });
        });

        group.MapPost("/{id:guid}/maintenance", async (
            Guid id,
            RecordMaintenanceRequest req,
            RecordMaintenanceHandler handler,
            ITenantContext tenant,
            CancellationToken ct) =>
        {
            await handler.HandleAsync(new RecordMaintenanceCommand(
                id, req.Technician, req.Notes,
                tenant.TenantId,
                req.ReasonForChange, req.SignedById, req.SignedByName), ct);
            return Results.NoContent();
        });

        group.MapPost("/{id:guid}/retire", async (
            Guid id,
            RetireAssetRequest req,
            RetireAssetHandler handler,
            ITenantContext tenant,
            CancellationToken ct) =>
        {
            await handler.HandleAsync(new RetireAssetCommand(
                id, req.Reason, tenant.TenantId,
                req.ReasonForChange, req.SignedById, req.SignedByName), ct);
            return Results.NoContent();
        });

        group.MapGet("/at-risk", async (
            IAssetQueryPort query,
            ITenantContext tenant,
            CancellationToken ct) =>
        {
            var assets = await query.GetAtRiskAsync(tenant.TenantId, ct);
            return Results.Json(assets,
                AssetManagerJsonContext.Default.IReadOnlyListAssetSummaryDto);
        });
    }
}
```
