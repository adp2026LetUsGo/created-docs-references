## FIX 1 — OUTBOX PATTERN (critical)

### Problem
When PredictOkEvent is published to Service Bus after prediction,
there is no guarantee of delivery if Service Bus is temporarily unavailable.
The prediction is saved to DB but the event is lost silently.
AssetManager and FinTracker never receive the event.

### Solution — Outbox Pattern (atomic with DB transaction)

**Step 1: Create outbox_messages table**

```sql
-- Add to existing Xinfer migration or create new:
-- 002_Xinfer_OutboxPattern.sql

CREATE TABLE outbox_messages (
    id              UUID        NOT NULL DEFAULT gen_random_uuid(),
    tenant_id       UUID        NOT NULL,
    event_type      VARCHAR(200) NOT NULL,
    payload_json    JSONB       NOT NULL,
    topic           VARCHAR(200) NOT NULL DEFAULT 'ahs.xinfer.events',
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    published_at    TIMESTAMPTZ NULL,
    error           TEXT        NULL,
    retry_count     INT         NOT NULL DEFAULT 0,
    CONSTRAINT pk_outbox_messages PRIMARY KEY (id)
);

CREATE INDEX ix_outbox_unpublished
    ON outbox_messages (created_at)
    WHERE published_at IS NULL;

-- RLS
ALTER TABLE outbox_messages ENABLE ROW LEVEL SECURITY;
CREATE POLICY xinfer_outbox_isolation ON outbox_messages
    USING (tenant_id = current_setting('app.current_tenant_id')::uuid);
```

**Step 5: Create OutboxPublisherService (BackgroundService)**

```csharp
// Infrastructure/Services/OutboxPublisherService.cs
namespace AHS.Cell.Xinfer.Infrastructure.Services;

public sealed class OutboxPublisherService(
    IServiceScopeFactory    scopeFactory,
    ServiceBusSender        sender,
    ILogger<OutboxPublisherService> log)
    : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            await PublishPendingAsync(ct);
            await Task.Delay(TimeSpan.FromSeconds(10), ct);
        }
    }

    private async Task PublishPendingAsync(CancellationToken ct)
    {
        await using var scope = scopeFactory.CreateAsyncScope();
        var db = scope.ServiceProvider.GetRequiredService<XinferDbContext>();

        // Dapper for performance — read unpublished outbox messages
        var conn = scope.ServiceProvider.GetRequiredService<IDbConnectionFactory>();
        await using var dbConn = await conn.CreateAsync(ct);

        var pending = (await dbConn.QueryAsync<OutboxMessage>("""
            SELECT id, tenant_id, event_type, payload_json, topic,
                   created_at, retry_count
            FROM outbox_messages
            WHERE published_at IS NULL
              AND retry_count < 5
            ORDER BY created_at
            LIMIT 50
            FOR UPDATE SKIP LOCKED
            """)).AsList();

        foreach (var msg in pending)
        {
            try
            {
                var sbMessage = new ServiceBusMessage(msg.PayloadJson)
                {
                    Subject        = msg.EventType,
                    ContentType    = "application/json",
                    MessageId      = msg.Id.ToString(),
                    CorrelationId  = msg.TenantId.ToString()
                };

                await sender.SendMessageAsync(sbMessage, ct);

                await dbConn.ExecuteAsync("""
                    UPDATE outbox_messages
                    SET published_at = NOW(), error = NULL
                    WHERE id = @id
                    """, new { id = msg.Id });

                log.LogInformation(
                    "Outbox: published {EventType} for tenant {TenantId}",
                    msg.EventType, msg.TenantId);
            }
            catch (Exception ex)
            {
                await dbConn.ExecuteAsync("""
                    UPDATE outbox_messages
                    SET retry_count = retry_count + 1, error = @error
                    WHERE id = @id
                    """, new { id = msg.Id, error = ex.Message });

                log.LogWarning(ex,
                    "Outbox: failed to publish {EventType} (retry {Retry})",
                    msg.EventType, msg.RetryCount + 1);
            }
        }
    }
}
```

## FIX 2 — OPERATIONAL HEALTH ENDPOINT

### Problem
`GET /health` returns generic ASP.NET health check (Healthy/Unhealthy).
AHS Hive cannot interpret Xinfer's operational state:
  - Is the ML model loaded?
  - Is retraining in progress?
  - Are there unprocessed outbox messages?

### Solution — Operational Health Response

**Step 1: Create XinferHealthService**

```csharp
// Infrastructure/Services/XinferHealthService.cs
namespace AHS.Cell.Xinfer.Infrastructure.Services;

public enum XinferLifecycleState
{
    Initializing,    // starting up, model loading
    Ready,           // fully operational
    Retraining,      // retraining in progress (predictions still served)
    Degraded,        // serving predictions with rule-based fallback (no active model)
    Draining,        // shutting down, no new requests accepted
    Maintenance      // manual maintenance mode
}

public sealed record XinferHealthDto(
    string               Status,           // "Healthy" | "Degraded" | "Unhealthy"
    XinferLifecycleState Lifecycle,
    string               ModelVersion,     // "v3" or "RuleBasedFallback"
    DateTimeOffset       ModelTrainedAt,
    int                  OutboxPending,    // unprocessed outbox messages
    bool                 IsRetraining,
    DateTimeOffset       CheckedAt
);

public class XinferHealthService(
    IModelRepository    models,
    IDbConnectionFactory db,
    ILogger<XinferHealthService> log)
{
    public async Task<XinferHealthDto> GetHealthAsync(
        Guid tenantId, CancellationToken ct)
    {
        var activeModel = await models.GetActiveAsync(ct);

        await using var conn = await db.CreateAsync(ct);
        var outboxPending = await conn.ExecuteScalarAsync<int>("""
            SELECT COUNT(*) FROM outbox_messages
            WHERE published_at IS NULL AND retry_count < 5
            """);

        var isRetraining = /* check BackgroundService state */ false;

        var lifecycle = (activeModel, isRetraining, outboxPending) switch
        {
            (null, _, _)           => XinferLifecycleState.Degraded,
            (_, true, _)           => XinferLifecycleState.Retraining,
            (_, _, > 100)          => XinferLifecycleState.Degraded,
            _                      => XinferLifecycleState.Ready
        };

        var status = lifecycle switch
        {
            XinferLifecycleState.Ready       => "Healthy",
            XinferLifecycleState.Retraining  => "Healthy",
            XinferLifecycleState.Degraded    => "Degraded",
            _                                => "Unhealthy"
        };

        return new XinferHealthDto(
            Status:         status,
            Lifecycle:      lifecycle,
            ModelVersion:   activeModel?.VersionNumber.ToString() ?? "RuleBasedFallback",
            ModelTrainedAt: activeModel?.TrainedAt ?? DateTimeOffset.MinValue,
            OutboxPending:  outboxPending,
            IsRetraining:   isRetraining,
            CheckedAt:      DateTimeOffset.UtcNow
        );
    }
}
```
