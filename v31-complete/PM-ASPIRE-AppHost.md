## SECTION 3 — PROGRAM.CS

```csharp
// AHS.AppHost/Program.cs
// AHS Hive — Single-command startup
// Run: dotnet run --project src/Infrastructure/AHS.AppHost
// Dashboard: http://localhost:15888

var builder = DistributedApplication.CreateBuilder(args);

// Infrastructure services
var postgres = builder.AddPostgres("ahs-postgres")
    .WithDataVolume("ahs-postgres-data")
    .WithPgAdmin();                         // pgAdmin at http://localhost:5050

var redis = builder.AddRedis("ahs-redis")
    .WithDataVolume("ahs-redis-data");

// AHS.Cell.Xinfer API
var xinfer = builder.AddProject<Projects.AHS_Cell_Xinfer_API>("xinfer-api")
    .WithReference(postgres)
    .WithReference(redis)
    .WithEnvironment("ASPNETCORE_ENVIRONMENT", "Development")
    .WithEnvironment("Xinfer__InputAdapter", "Local");

// AHS.Web.Hive (Blazor shell)
builder.AddProject<Projects.AHS_Web_Hive>("ahs-hive")
    .WithReference(xinfer)
    .WithReference(redis)
    .WithEnvironment("ASPNETCORE_ENVIRONMENT", "Development");

builder.Build().Run();
```
